  takeWhile p = foldr φ []

dropWhile :: (a -> Bool) -> [a] -> [a] Source #

リストの先頭から述語を満たす連続した要素を取り除く関数

  dropWhile p = para φ []
    where
      φ x (xs, ys)
        | p x       = ys
        | otherwise = x:xs

  para :: (a -> ([a], b) -> b) -> b -> [a] -> b
  para _  z []     = z
  para φ z (x:xs) = φ x (xs, para φ z xs)
