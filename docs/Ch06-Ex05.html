drop :: Int -> [a] -> [a] Source #

リストの先頭から指定した数の要素を落す

  drop :: Int -> [a] -> [a]
  drop 0 xs     = xs            -- (1)
  drop _ []     = []            -- (2)
  drop n (_:xs) = drop (n-1) xs -- (3)

    drop 3 [1,2,3,4,5]
  = { (3) }
    drop 2 [2,3,4,5]
  = { (3) }
    drop 1 [3,4,5]
  = { (3) }
    drop 0 [4,5]
  = { (1) }
    [4,5]

>>> drop 3 [1,2,3,4,5]
[4,5]

init :: [a] -> [a] Source #

リストの先頭部分

  init :: [a] -> [a]
  init [_]    = []          -- (1)
  init (x:xs) = x : init xs -- (2)

    init [1,2,3]
  = { (2) }
    1 : init [2,3]
  = { (2) }
    1 : (2 : init [3])
  = { (1) }
    1 : (2 : [])

>>> init [1,2,3]
[1,2]
