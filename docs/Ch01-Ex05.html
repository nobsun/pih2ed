qsort :: Ord a => [a] -> [a] Source #

  qsort []     = []                                    -- (1)
  qsort (x:xs) = qsort smaller ++ [x] ++ qsort larger  -- (2)
    where
      smaller = [a | a <- xs, a < x] -- <= を < に置き換える
      larger  = [b | b xs, b x]

qsortの定義で，<= を < に置き換えると，ピボットとと同値のピボット以外の要素を落してしまうことになる．

>>> qsort [2,2,3,1,1]
[1,2,3]
