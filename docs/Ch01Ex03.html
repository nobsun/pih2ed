product :: Num p => [p] -> p Source #

リストの要素の和

    product []     = 1           -- (1)
    product (n:ns) = n * sum ns  -- (2)
  
    product [2,3,4]
    = { [2,3,4] = (2:[3,4]) }
    product (2:[3,4])
    = { product の定義(2) }
    2 * product [3,4]
    = { [3,4] = (3:[4]) }
    2 * product (3:[4])
    = { product の定義(2) }
    2 * (3 * product [4])
    = { [4] = (4:[]) }
    2 * (3 * product (4:[]))
    = { product の定義(2) }
    2 * (3 * (4 * product []))
    = { product の定義(1) }
    2 * (3 * (4 * 1))
    = { 4 * 1 = 4 }
    2 * (3 * 4)
    = { 3 * 4 = 12 }
    2 * 12
    = { 2 * 12 = 12 }
