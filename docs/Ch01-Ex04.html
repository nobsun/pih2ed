qsort :: Ord a => [a] -> [a] Source #

降順に整列

  qsort []     = []                                    -- (1)
  qsort (x:xs) = qsort larger ++ [x] ++ qsort smaller  -- (2) qsort smaller と qsort larger を交換
    where
      smaller = [a | a <- xs, a <= x]
      larger  = [b | b xs, b x]

>>> qsort [3,5,1,4,2]
[5,4,3,2,1]
