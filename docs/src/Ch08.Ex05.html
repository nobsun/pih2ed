module Ch08.Ex05 where

{- |
  式
-}
data Expr
  = Val Int
  | Add Expr Expr
  deriving Show

{- |
  式の畳み込み

@
  folde f g e = case e of
    Val x     -> f x
    Add e1 e2 -> g (folde f g e1) (folde f g e2)
@
-}
folde :: (Int -> a) -> (a -> a -> a) -> Expr -> a
folde f g e = case e of
  Val x     -> f x
  Add e1 e2 -> g (folde f g e1) (folde f g e2)
