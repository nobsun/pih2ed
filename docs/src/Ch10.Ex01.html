リスト内包表記とプレリュード関数 @sequence_ :: [IO a] -> IO ()@ を用いて，@putStr :: String -> IO ()@ を再定義してください．
-}
module Ch10.Ex01 where

import Prelude hiding (putStr)

{- |
  再定義した @putStr@

@
putStr cs = sequence_ [ putChar c | c <- cs ]
@

>>> putStr "Hello, world!\n"
Hello, world!
-}
putStr :: String -> IO ()
putStr cs = sequence_ [ putChar c | c <- cs ]
