リスト内包表記を使って， 1 から 100 までの二乗の和 1^2 + 2^2 + ... + 100^2 を計算する式を考えてください．

module Ch05.Ex01 where

{- |
  2乗の和
-}
sumSquare :: [Int] -> Int
sumSquare xs = sum [ x ^ 2 | x <- xs ]
