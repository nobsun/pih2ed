他のプレリュード関数や演算子を使わずに，論理積 @&&@ に対する以下の定義を条件式を用いて形式化してください．

@
True  && b = b
False && _ = False
@

-}
module Ch04.Ex06 where

import Prelude hiding ((&&))

{- |
  連言の条件式を使った定式化

@
(&&) :: Bool -> Bool -> Bool
p && q = if p then q else False
@

-}
(&&) :: Bool -> Bool -> Bool
p && q = if p then q else False
