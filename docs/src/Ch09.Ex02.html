module Ch09.Ex02 where

import Ch09.CountDown hiding (choices)

{- |
  選択肢の生成

@
  choices xs = [ zs | ys <- subs xs, zs <- perms ys ]
@

>>> choices [1,2,3]
[[],[3],[2],[2,3],[3,2],[1],[1,3],[3,1],[1,2],[2,1],[1,2,3],[2,1,3],[2,3,1],[1,3,2],[3,1,2],[3,2,1]]
-}
choices :: [a] -> [[a]]
choices xs = [ zs | ys <- subs xs, zs <- perms ys ]
