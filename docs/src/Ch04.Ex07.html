以下のカリー化された関数の定義の意味をラムダ式を用いて形式化してください．

@
mult :: Int -> Int -> Int -> Int
mult x y z = x * y * z
@

-}
module Ch04.Ex07 where

{- |
  3つの整数の積

@
mult :: Int -> Int -> Int -> Int
mult = \ x -> \ y -> \ z -> x * y * z
@

>>> mult 2 3 4
24

-}
mult :: Int -> Int -> Int -> Int
mult = \ x -> \ y -> \ z -> x * y * z
