module Ch10.Ex03 where

import Ch10.Nim hiding (putBoard)

{- |
  ボードの表示

>>> putBoard initial
1: * * * * * 
2: * * * * 
3: * * * 
4: * * 
5: * 
-}
putBoard :: Board -> IO ()
putBoard board = sequence_ [ putRow r n | (r, n) <- zip [1..] board ]
