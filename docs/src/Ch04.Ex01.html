プレリュード関数を使って，長さが偶数のリストを半分ずつに分割する関数
@halve :: [a] -> ([a], [a])@ を定義してください．

-}
module Ch04.Ex01 where

{- |
  長さ偶数のリストを半分ずつに分割する．

>>> halve [1,2,3,4,5,6]
([1,2,3],[4,5,6])

-}
halve :: [a] -> ([a], [a])
halve = flip splitAt <*> (`div` 2) . length

{- --
