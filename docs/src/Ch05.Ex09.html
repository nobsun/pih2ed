{- |
  内積

@
 scalarproduct :: [Int] -> [Int] -> Int
 scalarproduct xs ys = sum [ x * y | (x, y) <- zip xs ys ]
@

>>> scalarproduct [1,2,3] [4,5,6]
32

scalarproduct :: [Int] -> [Int] -> Int
scalarproduct xs ys = sum [ x * y | (x, y) <- zip xs ys ]
