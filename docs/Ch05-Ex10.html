let2int :: Char -> Int Source #

文字を整数に変換 'a' -> 0，'b' -> 1，... ，'A' -> 26，'B' -> 27

>>> let2int 'a'
0
>>> let2int 'B'
27

int2let :: Int -> Char Source #

整数を文字に変換 0 -> 'a'，1 -> 'b'，...，26 -> 'A'，27 -> 'B'

>>> int2let 1
'b'
>>> int2let 26
'A'

shift :: Int -> Char -> Char Source #

アルファベット文字シフト

>>> shift 3 'a'
'd'
>>> shift 3 'z'
'c'
>>> shift (-3) 'c'
'z'
>>> shift 3 ' '
' '
>>> shift 3 'A'
'D'
>>> shift 3 'Z'
'C'
>>> shift (-3) 'C'
'Z'
